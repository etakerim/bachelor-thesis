155 void adi_dma_Init(void)
156 {
157  ADI_INT_STATUS_ALLOC();
158  ADI_ENTER_CRITICAL_REGION();
159 
160  if( false == pDMA_DevData->Initialized )
161  {
162  pDMA_DevData->Initialized = true;
163 
164  /* Enable the DMA Controller */
165  pADI_DMA0->CFG |= BITM_DMA_CFG_MEN;
166 
167  /* Set descriptor memory base pointer on DMA controller */
168  pADI_DMA0->PDBPTR = (uint32_t)pPrimaryCCD;
169 
170  /* Enable the DMA Error Interrupt */
171  NVIC_EnableIRQ(DMA_CHAN_ERR_IRQn);
172 
173  /* Reset per-channel, bitmapped control registers (W1C) */
174  const uint32_t w1r_value = (uint32_t) ((1 << NUM_DMA_CHANNELSn) - 1);
175  pADI_DMA0->RMSK_SET = w1r_value;
176  pADI_DMA0->EN_CLR = w1r_value;
177  pADI_DMA0->ALT_CLR = w1r_value;
178  pADI_DMA0->PRI_CLR = w1r_value;
179  pADI_DMA0->ERRCHNL_CLR = w1r_value;
180  pADI_DMA0->ERR_CLR = w1r_value;
181  pADI_DMA0->INVALIDDESC_CLR = w1r_value;
182  }
183 
184  ADI_EXIT_CRITICAL_REGION();
185 }

232 ADI_DMA_RESULT adi_dma_RegisterCallback (
233  DMA_CHANn_TypeDef const eChannelID,
234  ADI_CALLBACK const pfCallback,
235  void* const pCBParam
236  )
237 {
238  ADI_DMA_RESULT result = ADI_DMA_SUCCESS;
239 
240 #ifdef ADI_DEBUG
241  /* DMA must be initialized first */
242  if (false == pDMA_DevData->Initialized) {
243  result = ADI_DMA_ERR_NOT_INITIALIZED;
244  }else{
245  const size_t numChannelId = sizeof(pDMA_DevData->CallbackInfo) / sizeof(DMA_CHANNEL_CALLBACK_INFO);
246  if (numChannelId <= eChannelID)
247  {
248  result = ADI_DMA_ERR_INVALID_PARAMETER;
249  }
250  }
251  if (ADI_DMA_SUCCESS == result) /* if no errors previously detected */
252 #endif
253  {
254  /* eChannelID cannot be out of range by definition (we use DMA_CHANn_TypeDef) */
255  DMA_CHANNEL_CALLBACK_INFO * pChannel = &pDMA_DevData->CallbackInfo[eChannelID];
256 
257  /* Set the callback parameters */
258  pChannel->pfCallback = pfCallback; /* assign the pointer to a callback function */
259  pChannel->pCBParam = pCBParam; /* store the parameters to be used with the callback function */
260 
261  const uint32_t nChannelBit = (1u << eChannelID);
262  if (NULL != pfCallback) {
263  pDMA_DevData->ChannelsInUse |= nChannelBit; /* set the bit to mark the channel as "being used" */
264  }else{
265  pDMA_DevData->ChannelsInUse &= (~nChannelBit); /* clear the bit to mark the channel as "not being used" */
266  }
267  }
268  return result;
269 }

305 void DMA_Err_Int_Handler(void)
306 {
307  ISR_PROLOG()
308 
309  const uint32_t nErrClr = pADI_DMA0->ERR_CLR; /* get all the bits set in ERR_CLR */
310  const uint32_t nErrChnClr = pADI_DMA0->ERRCHNL_CLR; /* get all the bits set in ERRCHNL_CLR */
311  const uint32_t nInvdDescClr = pADI_DMA0->INVALIDDESC_CLR; /* get all the bits set in INVALIDDESC_CLR */
312 
313  /* if there are invalid channel descriptors or channel errors amongts the channels in use */
314  uint32_t functionsToBeCalled = pDMA_DevData->ChannelsInUse & (nErrChnClr | nInvdDescClr);
315 
316  if (functionsToBeCalled > 0u)
317  {
318  const uint32_t numBits = sizeof(uint32_t) << 3; /* maximum number of bits to be considered */
319  uint32_t nlz; /* number of leading zeroes in functionsToBeCalled */
320 
321  /* For all the bits set in functionsToBeCalled, starting from the MSB */
322  for (nlz = (uint32_t) ADI_CLZ(functionsToBeCalled); nlz < numBits; nlz = (uint32_t) ADI_CLZ(functionsToBeCalled))
323  {
324  const uint32_t bitSet = numBits - nlz - 1u; /* bit position in functionsToBeCalled */
325  const uint32_t selected_bit = ((uint32_t)1u << bitSet);
326  DMA_CHANNEL_CALLBACK_INFO* pChannel = &pDMA_DevData->CallbackInfo[bitSet];
327 
328  /* if there's a callback function to be called */
329  if (NULL != pChannel->pfCallback)
330  {
331  /* define the nature of the error: DMA bus error or else invalid descriptor */
332  uint32_t nEvent = ((nErrChnClr & selected_bit) != 0u)
333  ? (uint32_t)ADI_DMA_EVENT_ERR_BUS
334  : (uint32_t)ADI_DMA_EVENT_ERR_INVALID_DESCRIPTOR;
335 
336  /* report the error to the peripheral through the callback function */
337  pChannel->pfCallback (pChannel->pCBParam, nEvent, NULL );
338  }
339 
340  functionsToBeCalled &= ~selected_bit; /* clear bit in functionsToBeCalled */
341  }
342  }
343 
344  /* Clear the errors processed in the loop above */
345  pADI_DMA0->ERRCHNL_CLR = nErrChnClr; /* W1C: clear only all the bits set in nErrChnClr */
346  pADI_DMA0->INVALIDDESC_CLR = nInvdDescClr; /* W1C: clear only all the bits set in nInvdDescClr */
347  pADI_DMA0->ERR_CLR = nErrClr; /* W1C: clear only all the bits set in nErrClr */
348 
349  ISR_EPILOG()
350 }
